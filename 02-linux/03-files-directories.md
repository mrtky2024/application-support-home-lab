## Create a Directory (Create a new directory)
mkdir directory_name
## Create a File (Create a new file)
touch file_name
## Copy Files or Directories (Copy files or directories)
cp source destination
## Move or Rename Files and Directories (Move or rename files and directories)
mv source destination
## Remove a File (Delete a file)
rm file_name
## Remove a Directory and Its Contents (Delete a directory recursively)
rm -r directory_name
##Find Files or Directories (Search for files or directories)
find /path -name "file_name"
## Search Text in Files (Search for text inside files)
grep "text" file_name
## Search Text Recursively (Search for text in all files under a directory)
grep -rn "text" directory_name
